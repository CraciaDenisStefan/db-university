FROM `students`

Query 2:

SELECT count(*) AS crediti_10
FROM `courses`
WHERE `cfu` > 10;

Query 3:

SELECT * 
FROM `students` 
WHERE TIMESTAMPDIFF(YEAR,`date_of_birth` ,CURDATE()) > 30;
